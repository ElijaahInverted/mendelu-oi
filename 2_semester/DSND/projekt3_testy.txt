-- Test: neplatne id piva (ocekavana chyba ORA-20002)
BEGIN
  uprav_pivo(-1, 'NAZEV', 'Test');
EXCEPTION
  WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Chyba: ' || SQLERRM);
END;
/

-- Test: neplatny sloupec (ocekavan vypis "Takovýto sloupec neexistuje" + seznam sloupcu)

-- Test: platna aktualizace nazvu (rollback po kontrole)
EXEC uprav_pivo(1, 'NAZEV', 'Testovaci pivo');
SELECT nazev FROM piva WHERE id_piva = 1;
ROLLBACK;
SELECT nazev FROM piva WHERE id_piva = 1;

-- Test: pokus o poruseni integritniho omezeni (prazdny nazev - NOT NULL)
EXEC uprav_pivo(1, 'NAZEV', NULL);

-- ==== ##5## trg_restaurace_adresy ====

-- Test INSERT pres pohled (PSC 70200 = Ostrava, Moravskoslezsky kraj)
INSERT INTO restaurace_adresy(restaurace, obec, pop_cislo, ulice, orientacni_cislo, psc)
VALUES('Testovaci Hospoda', 'Ostrava', 1, 'Testovaci', '1a', 70200);
SELECT * FROM restaurace_adresy WHERE restaurace = 'Testovaci Hospoda';

-- Test UPDATE pres pohled
UPDATE restaurace_adresy
SET restaurace = 'Preimenovana Hospoda'
WHERE restaurace = 'Testovaci Hospoda';
SELECT * FROM restaurace_adresy WHERE restaurace = 'Preimenovana Hospoda';

-- Test DELETE pres pohled
DELETE FROM restaurace_adresy WHERE restaurace = 'Preimenovana Hospoda';
SELECT COUNT(*) FROM restaurace_adresy WHERE restaurace = 'Preimenovana Hospoda';

ROLLBACK;

-- Test INSERT s PSC mimo Moravskoslezsky kraj (ocekavana chyba ORA-20001)

-- ==== ##6## info_pivovaru ====

-- Zobrazit pivovar ID=1 a delku jeho nazvu pro overeni vetve
SELECT p.id_pivovaru, p.nazev, LENGTH(p.nazev) AS delka
FROM pivovary p
ORDER BY id_pivovaru
FETCH FIRST 10 ROWS ONLY;

-- Test: pivovar s krátkym nazvem (<= 6 znaku)
-- Najdi ID pivovaru s krátkym nazvem:
SELECT id_pivovaru, nazev, LENGTH(nazev) FROM pivovary WHERE LENGTH(nazev) <= 6 AND ROWNUM = 1;
-- Spust s nalezenim ID (napr. id = X):
-- SELECT info_pivovaru(X) FROM dual;

-- Test: obecne volani pro prvnich 5 pivovaru
SELECT p.id_pivovaru, p.nazev, LENGTH(p.nazev) AS delka, info_pivovaru(p.id_pivovaru) AS info
FROM pivovary p
