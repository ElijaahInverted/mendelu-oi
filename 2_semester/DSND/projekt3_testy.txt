-- Testovaci soubor pro Projekt 3 - PL/SQL (Varianta B)
-- Spustit pred kazdym blokem: SET SERVEROUTPUT ON SIZE UNLIMITED;
-- Bloky spoustet postupne - kazdy zacina === a konci prazdnym radkem.

-- ============================================================
-- BLOK 0: Kompilace vsech objektu
-- ============================================================
DECLARE
  v_invalid NUMBER := 0;
BEGIN
  DBMS_OUTPUT.PUT_LINE('=== BLOK 0: Stav kompilace objektu ===');
  FOR r IN (
    SELECT object_name, object_type, status
    FROM user_objects
    WHERE object_name IN (
      'PIVOVARY_V_KRAJI', 'INFO_RESTAURACE', 'UPRAV_PIVO',
      'RESTAURACE_ADRESY', 'TRG_RESTAURACE_ADRESY', 'INFO_PIVOVARU'
    )
    ORDER BY object_type, object_name
  ) LOOP
    IF r.status = 'VALID' THEN
      DBMS_OUTPUT.PUT_LINE('PASS: ' || r.object_type || ' ' || r.object_name || ' je VALID');
    ELSE
      DBMS_OUTPUT.PUT_LINE('FAIL: ' || r.object_type || ' ' || r.object_name || ' je ' || r.status);
      v_invalid := v_invalid + 1;
    END IF;
  END LOOP;
  IF v_invalid > 0 THEN
    DBMS_OUTPUT.PUT_LINE('>>> POZOR: ' || v_invalid || ' objekt(u) je INVALID - oprav pred dalsim testovanim <<<');
  ELSE
    DBMS_OUTPUT.PUT_LINE('>>> Vsechny objekty jsou VALID <<<');
  END IF;
END;

-- ============================================================
-- BLOK 1: pivovary_v_kraji
-- ============================================================
DECLARE
  v_kraj_valid   kraje.nazev%TYPE;
  v_kraj_invalid CONSTANT VARCHAR2(100) := 'Neexistujici Kraj XYZ 99999';

  PROCEDURE pass(p VARCHAR2) IS BEGIN DBMS_OUTPUT.PUT_LINE('PASS: ' || p); END;
  PROCEDURE fail(p VARCHAR2, r VARCHAR2 DEFAULT NULL) IS BEGIN
    DBMS_OUTPUT.PUT_LINE('FAIL: ' || p || CASE WHEN r IS NOT NULL THEN ' (' || r || ')' ELSE '' END);
  END;

BEGIN
  DBMS_OUTPUT.PUT_LINE('=== BLOK 1: pivovary_v_kraji ===');

  -- Nacti existujici kraj
  SELECT nazev INTO v_kraj_valid
  FROM (SELECT nazev FROM kraje
        JOIN smerovaci_cisla sc ON kraje.id_kraje = (
          SELECT o.id_kraje FROM obce o
          JOIN smerovaci_cisla s ON o.id_obce = s.id_obce
          JOIN adresy a ON s.psc = a.psc
          JOIN pivovary piv ON piv.id_adresy = a.id_adresy
          WHERE o.id_kraje = kraje.id_kraje AND ROWNUM = 1
        )
        ORDER BY nazev)
  WHERE ROWNUM = 1;

  -- Test 1: platny kraj nepada
  BEGIN
    pivovary_v_kraji(v_kraj_valid);
    pass('platny kraj bezi bez vyjimky [kraj: ' || v_kraj_valid || ']');
  EXCEPTION WHEN OTHERS THEN fail('platny kraj bezi bez vyjimky', SQLERRM); END;

  -- Test 2: neplatny kraj nepada (jen prazdny vystup)
  BEGIN
    pivovary_v_kraji(v_kraj_invalid);
    pass('neplatny kraj nepadne (prazdny vystup ocekavan)');
  EXCEPTION WHEN OTHERS THEN fail('neplatny kraj nepadne', SQLERRM); END;

END;

-- ============================================================
-- BLOK 2: info_restaurace
-- ============================================================
DECLARE
  v_id_valid   NUMBER;
  v_id_invalid CONSTANT NUMBER := -99999;

  PROCEDURE pass(p VARCHAR2) IS BEGIN DBMS_OUTPUT.PUT_LINE('PASS: ' || p); END;
  PROCEDURE fail(p VARCHAR2, r VARCHAR2 DEFAULT NULL) IS BEGIN
    DBMS_OUTPUT.PUT_LINE('FAIL: ' || p || CASE WHEN r IS NOT NULL THEN ' (' || r || ')' ELSE '' END);
  END;

BEGIN
  DBMS_OUTPUT.PUT_LINE('=== BLOK 2: info_restaurace ===');

  SELECT id_restaurace INTO v_id_valid
  FROM (SELECT id_restaurace FROM restaurace ORDER BY id_restaurace)
  WHERE ROWNUM = 1;

  -- Test 1: platne id nepada
  BEGIN
    info_restaurace(v_id_valid);
    pass('platne id bezi bez vyjimky [id: ' || v_id_valid || ']');
  EXCEPTION WHEN OTHERS THEN fail('platne id bezi bez vyjimky', SQLERRM); END;

  -- Test 2: neplatne id nepada (vyjimka zpracovana uvnitr procedury)
  BEGIN
    info_restaurace(v_id_invalid);
    pass('neplatne id nepada - vyjimka zachycena uvnitr');
  EXCEPTION WHEN OTHERS THEN fail('neplatne id nepada', SQLERRM); END;

  -- Test 3: pocet konkurentu neni zaporny
  DECLARE
    v_obec_id  obce.id_obce%TYPE;
    v_cnt_rest NUMBER;
    v_cnt_konk NUMBER;
  BEGIN
    SELECT sc.id_obce INTO v_obec_id
    FROM restaurace r
    JOIN adresy a ON r.id_adresy = a.id_adresy
    JOIN smerovaci_cisla sc ON a.psc = sc.psc
    WHERE r.id_restaurace = v_id_valid;

    SELECT COUNT(*) - 1 INTO v_cnt_konk
    FROM restaurace r2
    JOIN adresy a2 ON r2.id_adresy = a2.id_adresy
    JOIN smerovaci_cisla sc2 ON a2.psc = sc2.psc
    WHERE sc2.id_obce = v_obec_id;

    IF v_cnt_konk >= 0 THEN
      pass('pocet konkurentu je nezaporny (' || v_cnt_konk || ')');
    ELSE
      fail('pocet konkurentu je nezaporny', 'got: ' || v_cnt_konk);
    END IF;
  END;

  -- Test 4: delka separatoru odpovida delce hlavicky
  DECLARE
    v_nazev   restaurace.nazev%TYPE;
    v_obec    obce.nazev%TYPE;
    v_popisne adresy.cislo_popisne%TYPE;
    v_psc     adresy.psc%TYPE;
    v_kraj    kraje.nazev%TYPE;
    v_header  VARCHAR2(4000);
    v_sep_len NUMBER;
  BEGIN
    SELECT r.nazev, o.nazev, a.cislo_popisne, a.psc, k.nazev
    INTO v_nazev, v_obec, v_popisne, v_psc, v_kraj
    FROM restaurace r
    JOIN adresy a ON r.id_adresy = a.id_adresy
    JOIN smerovaci_cisla sc ON a.psc = sc.psc
    JOIN obce o ON sc.id_obce = o.id_obce
    JOIN kraje k ON o.id_kraje = k.id_kraje
    WHERE r.id_restaurace = v_id_valid;

    v_header := UPPER(v_nazev) || ' (' || v_obec || ', '
                || v_popisne || ', ' || v_psc || ', ' || v_kraj || ')';
    v_sep_len := LENGTH(v_header);

    IF v_sep_len > 0 THEN
      pass('delka separatoru je ' || v_sep_len || ' znaku (odpovida delce hlavicky)');
    ELSE
      fail('delka separatoru je nenulova');
    END IF;
  END;

END;

-- ============================================================
-- BLOK 3: uprav_pivo
-- ============================================================
DECLARE
  v_id_valid     NUMBER;
  v_id_invalid   CONSTANT NUMBER := -99999;
  v_nazev_before piva.nazev%TYPE;
  v_nazev_after  piva.nazev%TYPE;
  v_TEST_NAZEV   CONSTANT VARCHAR2(50) := 'ZmenenNazevTest999';

  PROCEDURE pass(p VARCHAR2) IS BEGIN DBMS_OUTPUT.PUT_LINE('PASS: ' || p); END;
  PROCEDURE fail(p VARCHAR2, r VARCHAR2 DEFAULT NULL) IS BEGIN
    DBMS_OUTPUT.PUT_LINE('FAIL: ' || p || CASE WHEN r IS NOT NULL THEN ' (' || r || ')' ELSE '' END);
  END;

BEGIN
  DBMS_OUTPUT.PUT_LINE('=== BLOK 3: uprav_pivo ===');

  SELECT id_piva INTO v_id_valid
  FROM (SELECT id_piva FROM piva ORDER BY id_piva)
  WHERE ROWNUM = 1;

  -- Test 1: neplatne id vyvola ORA-20002
  BEGIN
    uprav_pivo(v_id_invalid, 'NAZEV', 'test');
    fail('neplatne id vyvola ORA-20002 (vyjimka se nevyhodila)');
  EXCEPTION
    WHEN OTHERS THEN
      IF SQLCODE = -20002 THEN
        pass('neplatne id vyvola ORA-20002');
      ELSE
        fail('neplatne id vyvola ORA-20002', 'got: ' || SQLCODE || ' ' || SQLERRM);
      END IF;
  END;

  -- Test 2: neplatny sloupec nezpusobi pad
  BEGIN
    uprav_pivo(v_id_valid, 'SLOUPEC_KTERY_NEEXISTUJE_XYZ', 'hodnota');
    pass('neplatny sloupec nezpusobi vyjimku (chyba vypsana pres DBMS_OUTPUT)');
  EXCEPTION WHEN OTHERS THEN fail('neplatny sloupec nezpusobi vyjimku', SQLERRM); END;

  -- Test 3: neplatny sloupec nezmeni zadny radek
  DECLARE v_cnt NUMBER; BEGIN
    SELECT COUNT(*) INTO v_cnt FROM piva WHERE nazev = v_TEST_NAZEV;
    uprav_pivo(v_id_valid, 'SLOUPEC_NEEXISTUJE', v_TEST_NAZEV);
    DECLARE v_cnt2 NUMBER; BEGIN
      SELECT COUNT(*) INTO v_cnt2 FROM piva WHERE nazev = v_TEST_NAZEV;
      IF v_cnt2 = 0 THEN
        pass('neplatny sloupec nezmeni zadny radek');
      ELSE
        fail('neplatny sloupec nezmeni zadny radek', v_cnt2 || ' radku zmeneno');
      END IF;
    END;
  END;

  -- Test 4: platna aktualizace zmeni radek
  SELECT nazev INTO v_nazev_before FROM piva WHERE id_piva = v_id_valid;
  uprav_pivo(v_id_valid, 'NAZEV', v_TEST_NAZEV);
  SELECT nazev INTO v_nazev_after FROM piva WHERE id_piva = v_id_valid;
  IF v_nazev_after = v_TEST_NAZEV THEN
    pass('platna aktualizace NAZEV zmeni radek');
  ELSE
    fail('platna aktualizace NAZEV zmeni radek', 'got: ' || NVL(v_nazev_after, 'NULL'));
  END IF;
  ROLLBACK;

  -- Test 5: po rollbacku je puvodni hodnota
  SELECT nazev INTO v_nazev_after FROM piva WHERE id_piva = v_id_valid;
  IF v_nazev_after = v_nazev_before THEN
    pass('rollback obnovi puvodni hodnotu nazvu');
  ELSE
    fail('rollback obnovi puvodni hodnotu nazvu',
         'ocekavano=[' || v_nazev_before || '] got=[' || NVL(v_nazev_after, 'NULL') || ']');
  END IF;

  -- Test 6: NULL na NOT NULL sloupci je zachyceno uvnitr (nepropaguji vyjimku)
  BEGIN
    uprav_pivo(v_id_valid, 'NAZEV', NULL);
    pass('NULL na NOT NULL sloupci nepropaguji vyjimku (SQLERRM vypsano uvnitr)');
  EXCEPTION WHEN OTHERS THEN fail('NULL na NOT NULL sloupci nepropaguji vyjimku', SQLERRM); END;
  ROLLBACK;

END;

-- BLOK 5: trg_restaurace_adresy (INSTEAD OF trigger)

DECLARE
  v_psc_ms     smerovaci_cisla.psc%TYPE;
  v_psc_other  smerovaci_cisla.psc%TYPE;
  v_cnt        NUMBER;
  v_TEST_NAME  CONSTANT VARCHAR2(60) := '__TESTOVACI_HOSPODA_TRIGGER__';

  PROCEDURE pass(p VARCHAR2) IS BEGIN DBMS_OUTPUT.PUT_LINE('PASS: ' || p); END;
  PROCEDURE fail(p VARCHAR2, r VARCHAR2 DEFAULT NULL) IS BEGIN
    DBMS_OUTPUT.PUT_LINE('FAIL: ' || p || CASE WHEN r IS NOT NULL THEN ' (' || r || ')' ELSE '' END);
  END;

BEGIN
  DBMS_OUTPUT.PUT_LINE('=== BLOK 5: trg_restaurace_adresy ===');

  -- Najdi PSC v Moravskoslezskem kraji
  BEGIN
    SELECT sc.psc INTO v_psc_ms
    FROM (
      SELECT sc.psc FROM smerovaci_cisla sc
      JOIN obce o ON sc.id_obce = o.id_obce
      JOIN kraje k ON o.id_kraje = k.id_kraje
      WHERE k.nazev = 'Moravskoslezský kraj'
      ORDER BY sc.psc
    ) WHERE ROWNUM = 1;
  EXCEPTION WHEN NO_DATA_FOUND THEN v_psc_ms := NULL; END;

  -- Najdi PSC mimo Moravskoslezsky kraj
  BEGIN
    SELECT sc.psc INTO v_psc_other
    FROM (
      SELECT sc.psc FROM smerovaci_cisla sc
      JOIN obce o ON sc.id_obce = o.id_obce
      JOIN kraje k ON o.id_kraje = k.id_kraje
      WHERE k.nazev != 'Moravskoslezský kraj'
      ORDER BY sc.psc
    ) WHERE ROWNUM = 1;
  EXCEPTION WHEN NO_DATA_FOUND THEN v_psc_other := NULL; END;

  SAVEPOINT sp_trigger_start;

  -- Test 1: INSERT s platnym PSC
  IF v_psc_ms IS NULL THEN
    DBMS_OUTPUT.PUT_LINE('SKIP: zadne PSC v Moravskoslezskem kraji - preskakuji INSERT testy');
  ELSE
    BEGIN
      INSERT INTO restaurace_adresy(restaurace, obec, pop_cislo, ulice, orientacni_cislo, psc)
      VALUES(v_TEST_NAME, 'TestObec', 1, 'Testovaci', '1t', v_psc_ms);

      SELECT COUNT(*) INTO v_cnt FROM restaurace_adresy WHERE restaurace = v_TEST_NAME;
      IF v_cnt = 1 THEN pass('INSERT pres pohled: radek videt v pohledu');
      ELSE fail('INSERT pres pohled: radek videt v pohledu', 'count=' || v_cnt); END IF;

      SELECT COUNT(*) INTO v_cnt FROM restaurace WHERE nazev = v_TEST_NAME;
      IF v_cnt = 1 THEN pass('INSERT pres pohled: radek existuje v tabulce restaurace');
      ELSE fail('INSERT pres pohled: radek existuje v tabulce restaurace'); END IF;

      -- Test 2: UPDATE nazvu
      UPDATE restaurace_adresy
      SET restaurace = v_TEST_NAME || '_UPD'
      WHERE restaurace = v_TEST_NAME;

      SELECT COUNT(*) INTO v_cnt FROM restaurace_adresy WHERE restaurace = v_TEST_NAME || '_UPD';
      IF v_cnt = 1 THEN pass('UPDATE pres pohled: nazev zmenen');
      ELSE fail('UPDATE pres pohled: nazev zmenen'); END IF;

      -- Test 3: UPDATE ulice
      UPDATE restaurace_adresy
      SET ulice = 'UpravenaUlice'
      WHERE restaurace = v_TEST_NAME || '_UPD';

      SELECT COUNT(*) INTO v_cnt
      FROM restaurace_adresy WHERE restaurace = v_TEST_NAME || '_UPD' AND ulice = 'UpravenaUlice';
      IF v_cnt = 1 THEN pass('UPDATE pres pohled: ulice zmenena');
      ELSE fail('UPDATE pres pohled: ulice zmenena'); END IF;

      -- Test 4: DELETE
      DELETE FROM restaurace_adresy WHERE restaurace = v_TEST_NAME || '_UPD';
      SELECT COUNT(*) INTO v_cnt FROM restaurace WHERE nazev = v_TEST_NAME || '_UPD';
      IF v_cnt = 0 THEN pass('DELETE pres pohled: radek odstranen z restaurace');
      ELSE fail('DELETE pres pohled: radek odstranen z restaurace', 'stale existuje'); END IF;

    EXCEPTION WHEN OTHERS THEN
      fail('INSERT/UPDATE/DELETE pres pohled', SQLERRM);
    END;
  END IF;

  ROLLBACK TO sp_trigger_start;

  -- Test 5: INSERT s PSC mimo Moravskoslezsky kraj vyvola chybu
  IF v_psc_other IS NULL THEN

-- ============================================================
-- BLOK 6: info_pivovaru
-- ============================================================
DECLARE
  v_id_short  NUMBER;   -- nazev <= 6 znaku
  v_id_medium NUMBER;   -- nazev 7-10 znaku
  v_id_long   NUMBER;   -- nazev > 10 znaku
  v_result    VARCHAR2(4000);

  PROCEDURE pass(p VARCHAR2) IS BEGIN DBMS_OUTPUT.PUT_LINE('PASS: ' || p); END;
  PROCEDURE fail(p VARCHAR2, r VARCHAR2 DEFAULT NULL) IS BEGIN
    DBMS_OUTPUT.PUT_LINE('FAIL: ' || p || CASE WHEN r IS NOT NULL THEN ' (' || r || ')' ELSE '' END);
  END;

  FUNCTION count_semicolons(p_str VARCHAR2) RETURN NUMBER IS
    v_n NUMBER := 0; v_p NUMBER := 1;
  BEGIN
    LOOP v_p := INSTR(p_str, ';', v_p);
      EXIT WHEN v_p = 0; v_n := v_n + 1; v_p := v_p + 1;
    END LOOP;
    RETURN v_n;
  END;

  PROCEDURE check_branch(p_id NUMBER, p_label VARCHAR2, p_expected_semis NUMBER) IS
    v_res VARCHAR2(4000);
  BEGIN
    IF p_id IS NULL THEN
      DBMS_OUTPUT.PUT_LINE('SKIP: zadny pivovar s nazvem ' || p_label);
      RETURN;
    END IF;
    v_res := info_pivovaru(p_id);
    IF v_res IS NULL THEN
      fail(p_label || ': vraci nenulovy retezec', 'got NULL');
    ELSIF count_semicolons(v_res) != p_expected_semis THEN
      fail(p_label || ': ' || p_expected_semis || ' strednik(u)',
           'got ' || count_semicolons(v_res) || ' v [' || v_res || ']');
    ELSIF SUBSTR(v_res, -1) != '.' THEN
      fail(p_label || ': konci teckou', 'got [' || SUBSTR(v_res, -1) || ']');
    ELSE
      pass(p_label || ': format OK [' || v_res || ']');
    END IF;
  END;

BEGIN
  DBMS_OUTPUT.PUT_LINE('=== BLOK 6: info_pivovaru ===');

  -- Test 1: neexistujici id vraci NULL
  v_result := info_pivovaru(-99999);
  IF v_result IS NULL THEN pass('neexistujici id vraci NULL');
  ELSE fail('neexistujici id vraci NULL', 'got: ' || v_result); END IF;

  -- Nacti ID podle delky nazvu
  BEGIN SELECT id_pivovaru INTO v_id_short FROM
    (SELECT id_pivovaru FROM pivovary WHERE LENGTH(nazev) <= 6 ORDER BY id_pivovaru) WHERE ROWNUM=1;
  EXCEPTION WHEN NO_DATA_FOUND THEN v_id_short := NULL; END;

  BEGIN SELECT id_pivovaru INTO v_id_medium FROM
    (SELECT id_pivovaru FROM pivovary WHERE LENGTH(nazev) BETWEEN 7 AND 10 ORDER BY id_pivovaru) WHERE ROWNUM=1;
  EXCEPTION WHEN NO_DATA_FOUND THEN v_id_medium := NULL; END;

  BEGIN SELECT id_pivovaru INTO v_id_long FROM
    (SELECT id_pivovaru FROM pivovary WHERE LENGTH(nazev) > 10 ORDER BY id_pivovaru) WHERE ROWNUM=1;
  EXCEPTION WHEN NO_DATA_FOUND THEN v_id_long := NULL; END;

  -- Test 2: kratky nazev (<= 6 znaku) -> 1 strednik, konci teckou
  check_branch(v_id_short, 'kratky nazev (<= 6 znaku) -> 1 strednik', 1);

  -- Test 3: stredni nazev (7-10 znaku) -> 2 stredniki, konci teckou
  check_branch(v_id_medium, 'stredni nazev (7-10 znaku) -> 2 stredniki', 2);

  -- Test 4: dlouhy nazev (> 10 znaku) -> 2 stredniki, konci teckou
  check_branch(v_id_long, 'dlouhy nazev (> 10 znaku) -> 2 stredniki', 2);

  -- Test 5: vysledek pro dlouhy nazev obsahuje cislo (objem) jako treti cast
  IF v_id_long IS NOT NULL THEN
    v_result := info_pivovaru(v_id_long);
    IF v_result IS NOT NULL AND count_semicolons(v_result) = 2 THEN
      DECLARE
        v_third VARCHAR2(200);
        v_pos1  NUMBER;
        v_pos2  NUMBER;
      BEGIN
        v_pos1 := INSTR(v_result, ';', 1, 1);
        v_pos2 := INSTR(v_result, ';', 1, 2);
        v_third := SUBSTR(v_result, v_pos2 + 1, LENGTH(v_result) - v_pos2 - 1);
        IF REGEXP_LIKE(v_third, '^\d+(\.\d+)?$') THEN
          pass('dlouhy nazev: treti cast je cislo (objem v litrech) [' || v_third || ']');
        ELSE
          fail('dlouhy nazev: treti cast je cislo', 'got [' || v_third || ']');
        END IF;
      END;
    END IF;
  END IF;

  -- Test 6: stredni nazev obsahuje cislo jako treti cast (pocet druhu)
  IF v_id_medium IS NOT NULL THEN
    v_result := info_pivovaru(v_id_medium);
    IF v_result IS NOT NULL AND count_semicolons(v_result) = 2 THEN
      DECLARE
        v_third VARCHAR2(200);
        v_pos1  NUMBER;
        v_pos2  NUMBER;
      BEGIN
        v_pos1 := INSTR(v_result, ';', 1, 1);
        v_pos2 := INSTR(v_result, ';', 1, 2);
        v_third := SUBSTR(v_result, v_pos2 + 1, LENGTH(v_result) - v_pos2 - 1);
        IF REGEXP_LIKE(v_third, '^\d+$') THEN
          pass('stredni nazev: treti cast je cele cislo (pocet druhu) [' || v_third || ']');
        ELSE
          fail('stredni nazev: treti cast je cele cislo', 'got [' || v_third || ']');
        END IF;
      END;
    END IF;
  END IF;

END;
